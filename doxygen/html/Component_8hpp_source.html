168 void animateAndUpdate(SpriteRenderer* renderer, Transforms* transforms, sf::Time frameDuration, const std::vector<std::string>& frameNames) {
169 sf::Time currentTime = frameClock.getElapsedTime();
170 sf::Time elapsedTime = currentTime - lastFrameTime;
171
172 if (elapsedTime >= frameDuration) {
173 lastFrameTime = currentTime;
174 if (currentFrameIndex < frameNames.size()) {
175 currentFrameName = frameNames[currentFrameIndex];
176 // std::cout << "frame index : " << currentFrameIndex << std::endl;
177 renderer->setSprite(spritesManager.getSprite(frameNames[currentFrameIndex]));
178 renderer->sprite->setPosition(transforms->position);
179 renderer->sprite->setScale(transforms->scale);
180 currentFrameIndex++;
181 }
182 }
183
184 // reset the currentFrameIndex to create loop
185 if (currentFrameIndex >= frameNames.size()) {
186 currentFrameIndex = 0;
187 }
188 }
189
190 std::string get_frame() {
191 return currentFrameName;
192 }
193
194 private:
195 SpriteManager& spritesManager;
196 sf::Clock frameClock;
197 sf::Time frameDuration;
198 sf::Time lastFrameTime;
199 std::vector<std::string> frameNames;
200 int currentFrameIndex;
201 std::string currentFrameName;
